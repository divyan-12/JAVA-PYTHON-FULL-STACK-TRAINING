class Solution {
    public int countKeyChanges(String s) {

        int count = 0;

        // Convert the entire string to lowercase
        s = s.toLowerCase();

        // Compare adjacent characters
        for (int i = 1; i < s.length(); i++) {
            if (s.charAt(i) != s.charAt(i - 1)) {
                count++;
            }
        }

        return count;
    }
}
